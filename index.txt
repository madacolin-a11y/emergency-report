// ============================================================
// 設定區：請填入你的資訊
// ============================================================
const GAS_URL = 'https://script.google.com/macros/s/AKfycbxWbFHRmVzggyCbLlvg_PGdLPeV7TfSo0A9JHugCBmtcKDzIVD9wNRm89JxYguagJRkPQ/exec';  // ← 貼上第一步取得的 GAS 網址
const LIFF_ID = 'YOUR_LIFF_ID';           // ← 第三步建立 LIFF 後填入

// ============================================================
// 全域變數
// ============================================================
let lineUserId = null;

// ============================================================
// 初始化
// ============================================================
async function init() {
  // 設定發生時間預設為現在
  const now = new Date();
  now.setMinutes(now.getMinutes() - now.getTimezoneOffset());
  document.getElementById('incidentTime').value = now.toISOString().slice(0, 16);
  
  // 載入旅行社名單
  await loadAgencies();
  
  // 初始化 LIFF（取得 LINE 使用者資訊）
  try {
    await liff.init({ liffId: LIFF_ID });
    if (liff.isLoggedIn()) {
      const profile = await liff.getProfile();
      lineUserId = profile.userId;
      console.log('LINE userId:', lineUserId);
    }
  } catch (err) {
    console.warn('LIFF 初始化失敗（可能在瀏覽器測試中）：', err);
  }
  
  document.getElementById('loading').style.display = 'none';
  document.getElementById('mainForm').style.display = 'block';
}

// ============================================================
// 從試算表載入旅行社名單
// ============================================================
async function loadAgencies() {
  try {
    const res = await fetch(`${GAS_URL}?action=getAgencies`);
    const data = await res.json();
    const select = document.getElementById('agencyName');
    
    data.agencies.forEach(name => {
      const opt = document.createElement('option');
      opt.value = name;
      opt.textContent = name;
      select.appendChild(opt);
    });
  } catch (err) {
    console.warn('旅行社名單載入失敗，請手動填寫', err);
    // 改為文字輸入作為 fallback
    const select = document.getElementById('agencyName');
    const input = document.createElement('input');
    input.type = 'text';
    input.id = 'agencyName';
    input.placeholder = '請輸入旅行社名稱';
    select.parentNode.replaceChild(input, select);
  }
}

// ============================================================
// 通報類別選擇
// ============================================================
function selectType(type) {
  document.querySelectorAll('.type-btn').forEach(btn => {
    btn.className = 'type-btn';
  });
  event.target.className = `type-btn selected-${type}`;
  document.getElementById('reportType').value = type;
}

// ============================================================
// GPS 定位
// ============================================================
function getGPS() {
  const btn = document.getElementById('gpsBtn');
  btn.textContent = '定位中…';
  btn.disabled = true;
  
  navigator.geolocation.getCurrentPosition(
    pos => {
      const coord = `${pos.coords.latitude.toFixed(6)}, ${pos.coords.longitude.toFixed(6)}`;
      document.getElementById('gpsCoord').value = coord;
      btn.textContent = '✅ 已定位';
      btn.style.background = '#27ae60';
    },
    err => {
      alert('定位失敗，請手動填寫地點。\n錯誤：' + err.message);
      btn.textContent = '📍 定位';
      btn.disabled = false;
    },
    { enableHighAccuracy: true, timeout: 10000 }
  );
}

// ============================================================
// 表單送出
// ============================================================
async function submitForm() {
  // 驗證必填欄位
  const required = [
    ['reportType', '通報類別'],
    ['tourCode', '團號'],
    ['agencyName', '旅行社名稱'],
    ['tourStartDate', '旅遊開始日期'],
    ['guideName', '導遊/領隊姓名'],
    ['reporterName', '通報人姓名'],
    ['phone', '聯絡電話'],
    ['incidentTime', '發生時間'],
    ['location', '發生地點'],
    ['description', '事情經過'],
    ['currentStatus', '目前處理情形'],
  ];
  
  for (const [id, label] of required) {
    const el = document.getElementById(id);
    if (!el || !el.value.trim()) {
      alert(`⚠️ 請填寫「${label}」`);
      el && el.focus();
      return;
    }
  }
  
  // 電話格式驗證
  const phone = document.getElementById('phone').value;
  if (!/^[\d\-+() ]{8,15}$/.test(phone)) {
    alert('⚠️ 請填寫正確的聯絡電話格式');
    return;
  }
  
  const submitBtn = document.getElementById('submitBtn');
  submitBtn.disabled = true;
  submitBtn.textContent = '送出中…';
  
  const payload = {
    reportType:    document.getElementById('reportType').value,
    tourCode:      document.getElementById('tourCode').value.trim(),
    agencyName:    document.getElementById('agencyName').value,
    tourStartDate: document.getElementById('tourStartDate').value,
    tourEndDate:   document.getElementById('tourEndDate').value,
    guideName:     document.getElementById('guideName').value.trim(),
    reporterName:  document.getElementById('reporterName').value.trim(),
    phone:         phone,
    incidentTime:  document.getElementById('incidentTime').value,
    location:      document.getElementById('location').value.trim(),
    gps:           document.getElementById('gpsCoord').value,
    description:   document.getElementById('description').value.trim(),
    currentStatus: document.getElementById('currentStatus').value.trim(),
    userId:        lineUserId,
  };
  
  try {
    const res = await fetch(GAS_URL, {
      method: 'POST',
      headers: { 'Content-Type': 'text/plain' }, // GAS 需要 text/plain 避免 CORS 問題
      body: JSON.stringify(payload),
    });
    const result = await res.json();
    
    if (result.status === 'success') {
      showStatus('success', '✅ 通報成功！觀光署已收到您的回報，謝謝。');
      submitBtn.textContent = '✅ 已送出';
      // 5 秒後關閉 LIFF
      setTimeout(() => { try { liff.closeWindow(); } catch(e) {} }, 5000);
    } else {
      throw new Error(result.message);
    }
  } catch (err) {
    showStatus('error', '❌ 送出失敗：' + err.message + '\n請重試或來電通報。');
    submitBtn.disabled = false;
    submitBtn.textContent = '📤 送出通報';
  }
}

function showStatus(type, msg) {
  const el = document.getElementById('statusMsg');
  el.className = `status-msg ${type}`;
  el.textContent = msg;
  el.style.display = 'block';
  el.scrollIntoView({ behavior: 'smooth' });
}

// 啟動
init();
</script>
</body>
</html>

```
## 五、第三步：串接 LINE Bot 與 LIFF

### 3-1 建立 LINE Bot

1. 前往 [LINE Developers Console](https://developers.line.biz/)
2. 建立 Provider → 建立 **Messaging API Channel**
3. 在 Channel 設定頁面，取得：
   - **Channel Access Token**（填入 GAS 程式碼的 `CONFIG.LINE_CHANNEL_ACCESS_TOKEN`）

### 3-2 建立 LIFF App

1. 在同一個 Channel 下，點選 **LIFF** 分頁
2. 點「Add」新增 LIFF App：
   - Size：**Full**（全螢幕）
   - Endpoint URL：填入你部署 HTML 的 GitHub Pages 網址
3. 取得 **LIFF ID**（格式如 `1234567890-AbCdEfGh`），填入 HTML 的 `LIFF_ID` 變數

### 3-3 設定圖文選單

在 LINE Official Account Manager 建立圖文選單，設定按鈕動作為：
```
動作類型：開啟 LIFF
LIFF URL：https://liff.line.me/你的LIFF_ID
